 25 | ||| Compute an accept ratio using the previous MH result and the newly proposed MH result
 26 | export
 27 | accept
 28 |    : Addr               -- ^ proposal sample address
 29 |   -> (STrace, LPTrace)  -- ^ previous sample and log-prob traces
 30 |   -> (STrace, LPTrace)  -- ^ new sample and log-prob traces
 31 |   -> Double             -- ^ acceptance ratio
 32 | accept x0 (strace, lptrace) (strace', lptrace') =
 33 |   let s_addrs  = keySet strace
 34 |       s_addrs' = keySet strace'
 35 |       dom_log  = log (cast $ length $ SortedSet.toList s_addrs) - log (cast $ length $ SortedSet.toList s_addrs')
 36 |       sampled  = singleton x0 `union` (s_addrs  `difference` s_addrs')
 37 |       sampled' = singleton x0 `union` (s_addrs' `difference` s_addrs)
 38 |       logα     = foldl (\logα, k => logα + fromMaybe 0 (lookup k lptrace))
 39 |                         0 (keySet lptrace `difference` sampled)
 40 |       logα'    = foldl (\logα, k => logα + fromMaybe 0 (lookup k lptrace'))
 41 |                         0 (keySet lptrace' `difference` sampled')
 42 |   in  min 1.0 (exp (dom_log + logα' - logα))

 44 | ||| Handler for reusing samples (and generating new ones when necessary)
 45 | export
 46 | handleSample : STrace -> Prog [Sample] a -> Sampler (a, STrace)
 47 | handleSample strace (Val x)   = pure (x, strace)
 48 | handleSample strace (Op op k) with (prj1 op)
 49 |   _ | (MkSample d addr) with (lookup addr strace)
 50 |     _ | Just r  = do sample d r >>= (handleSample strace . k)
 51 |     _ | Nothing = do r <- random
 52 |                      sample d r >>= (handleSample (insert addr r strace) . k)
 53 |
 54 | ||| Handler for recording log-probabilities
 55 | export
 56 | handleObserve : Elem Observe es => Prog es a -> Prog (es - Observe) (a, LPTrace)
 57 | handleObserve = loop empty
 58 |   where
 59 |     loop : LPTrace -> Prog es a -> Prog (es - Observe) (a, LPTrace)
 60 |     loop lptrace (Val x)   = pure (x, lptrace)
 61 |     loop lptrace (Op op k) = case discharge op of
 62 |       Left op'                   => Op op' (loop lptrace . k)
 63 |       Right (MkObserve d y addr) => loop (insert addr (logProb d y) lptrace) (k y)

 65 | ||| Handler for one program execution under MH
 66 | export
 67 | handleMH : STrace -> Prog [Observe, Sample] a -> Sampler ((a, LPTrace), STrace)
 68 | handleMH strace = handleSample strace . handleObserve

 70 | ||| Perform one iteration of MH
 71 | export
 72 | mhStep
 73 |    : Prog [Observe, Sample] a
 74 |   -> List1 ((a, LPTrace), STrace)            -- ^ trace of previous MH results
 75 |   -> Sampler (List1 ((a, LPTrace), STrace))  -- ^ updated trace of MH results
 76 | mhStep prog trace = do
 77 |   let ((a, lptrace), strace) = head trace
 78 |       addrs : List Addr
 79 |       addrs = keys strace
 80 |       sample_sites : PrimDist Addr
 81 |       sample_sites = Discrete $ Vect.fromList $ (("", 0), 0.0) :: (map (, 1.0 / cast (length addrs)) addrs)
 82 |
 83 |   x0  <- sample sample_sites !random
 84 |   ((a', lptrace'), strace') <- handleMH (insert x0 !random strace) prog
 85 |
 86 |   let accept_ratio = accept x0 (strace, lptrace) (strace', lptrace')
 87 |   pure $ if accept_ratio > !random
 88 |           then ((a', lptrace'), strace') ::: List1.forget trace
 89 |           else trace
 90 |
 91 | ||| Perform MH for n iterations
 92 | export
 93 | mhInternal
 94 |    : Nat                                    -- ^ number of MH iterations
 95 |   -> Prog [Observe, Sample] a
 96 |   -> Sampler (List ((a, LPTrace), STrace))  -- ^ trace of MH results
 97 | mhInternal n prog = do
 98 |   mh_out_0 <- handleMH SortedMap.empty prog
 99 |   List1.forget <$> foldl (>=>) pure (replicate n (mhStep prog)) (mh_out_0 ::: [])
100 |
101 | ||| Top-level wrapper for MH on a model
102 | export
103 | mh : Nat                          -- ^ number of MH iterations
104 |   -> Model env [] a               -- ^ model
105 |   -> Env env                      -- ^ input model environment
106 |   -> Sampler (List (Env env))     -- ^ output model environments
107 | mh n model env_in = do
108 |   let prog : Prog [Observe, Sample] (a, Env env)
109 |       prog = handleCore env_in model
110 |   mh_trace <- mhInternal n prog
111 |   pure (map (snd . fst . fst) mh_trace)
