 20 | data Record : List (String, Type) -> Type where
 21 |   Empty : Record []
 22 |   Cons : (key : String) -> (val : a) -> Record rows -> Record ((key, a) :: rows)
 23 |
 24 | update : (key : String) -> (new : a) -> Record k -> {auto prf : Elem (key, a) k} -> Record k
 25 | update key new (Cons key val r) {prf = Here} = Cons key new r
 26 | update key new (Cons y val r) {prf = (There later)} = Cons y val $ update key new r
 27 |

 29 | lookup key (Cons key val x) {prf = Here} = val
 30 | lookup key (Cons x val y) {prf = (There later)} = lookup key y

 32 | ||| SIR transition models
 33 | transSI : Elem ("s", Nat) popl => Elem ("i", Nat) popl => Elem ("r", Nat) popl =>
 34 |   TransModel env es Double (Record popl)
 35 | transSI beta pop  = do
 36 |   let (s_0, i_0, r_0) : (Nat, Nat, Nat) = (lookup "s" pop,  lookup "i" pop, lookup "r" pop)
 37 |   dN_SI <- binomial' {env} s_0 (1 - exp ((-beta * cast i_0) / cast (s_0 + i_0 + r_0))) "dN_SI"
 38 |
 39 |   pure $ update "s" (minus s_0 dN_SI) (update "i" (i_0 + dN_SI) pop)
 40 |
 41 | transIR : Elem ("i", Nat) popl => Elem ("r", Nat) popl =>
 42 |   TransModel env es Double (Record popl)
 43 | transIR gamma pop = do
 44 |   let (i_0, r_0) =  (lookup "i" pop,  lookup "r" pop)
 45 |   dN_IR <- binomial' {env} i_0 (1 - exp (-gamma)) "dN_IR"
 46 |
 47 |   pure $ update "r" (r_0 + dN_IR) (update "i" (minus i_0 dN_IR) pop)
 48 |
 49 | transSIR : Elem ("s", Nat) popl => Elem ("i", Nat) popl => Elem ("r", Nat) popl =>
 50 |   TransModel env es (Double, Double) (Record popl)
 51 | transSIR (beta, gamma) pop = do
 52 |   pop' <- (transSI beta >=> transIR gamma) pop
 53 |   pure pop'

132 | ||| MH inference on the SIR model, via monad bayes.
