 50 | ||| Distribution over the topics in a document, over the distribution of words in a topic
 51 | topicModel :
 52 |      {vocab_size_pred : Nat}
 53 |   -> (vocab    : Vect (S vocab_size_pred) String)
 54 |   -> (n_topics_pred : Nat)
 55 |   -> (doc_size : Nat)
 56 |   -> (Observable env "φ" (Vect (S vocab_size_pred) Double),
 57 |       Observable env "θ" (Vect (S n_topics_pred) Double),
 58 |       Observable env "w" String)
 59 |   => Model env es (Vect doc_size String)
 60 | topicModel vocab n_topics_pred doc_size = do
 61 |   -- Generate distribution over words for each topic
 62 |   topic_word_ps <- replicateM (S n_topics_pred) (topicWordPrior vocab)
 63 |   -- Generate distribution over topics for a given document
 64 |   doc_topic_ps  <- docTopicPrior n_topics_pred
 65 |   -- Use the above distributions to generate words for a document
 66 |   words         <- replicateM doc_size (do z <- categorical' doc_topic_ps "z"
 67 |                                            let word_ps = index z topic_word_ps
 68 |                                            wordDist vocab word_ps)
 69 |   pure words
